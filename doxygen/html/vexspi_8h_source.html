75 /** @details
76  * Format of the joystick component of the SPI receive packet
77  * The reserved bytes are really used for other purposes
78  * but we keep them here in V1.00
79  */
80 typedef struct _jsdata {
81  char Ch1; ///< Analog control channel 1
82  char Ch2; ///< Analog control channel 2
83  char Ch3; ///< Analog control channel 3
84  char Ch4; ///< Analog control channel 4
85  char acc_y; ///< Accelerometer y data
86  char acc_x; ///< Accelerometer x data
87  char acc_z; ///< Accelerometer z data - not used
88  unsigned char btns[2]; ///< data for the 12 buttons
89  unsigned char res[2]; ///< reserved
90  } jsdata;

93 /** @brief SPI transmit data packet */
94 /*-----------------------------------------------------------------------------*/
95 /** @details
96  * Format of a SPI transmit packet - 16 words
97  */
98 typedef union _spiTxPacket {
99  struct _spiTxPak {
100  unsigned char h1; ///< Header byte one
101  unsigned char h2; ///< Header byte two
102  unsigned char reserved[4]; ///< unknown data
103  unsigned char motor[8]; ///< motor pwm data
104  unsigned char pad[15]; ///< not used
105  unsigned char rev_lsb; ///< revision of user code lsb
106  unsigned char rev_msb; ///< revision of user code msb
107  unsigned char id; ///< message id
108  } pak; ///< access spiTxPacket as named variables
109 
110  unsigned char data[32]; ///< access spiTxPacket as an array of char
111 } spiTxPacket;

113 /*-----------------------------------------------------------------------------*/
114 /** @brief SPI receive data packet */
115 /*-----------------------------------------------------------------------------*/
116 /** @details
117  * Format of a SPI receive packet - 16 words
118  */
119 typedef union _spiRxPacket {
120  struct _spiRxPak {
121  unsigned char h1; ///< Header byte one
122  unsigned char h2; ///< Header byte two
123  unsigned char res1; ///< unknown
124  unsigned char ctl; ///< status and control byte
125  unsigned char batt1; ///< main battery level, 59mV per bit
126  unsigned char batt2; ///< backup battery level, 59mV per bit
127  jsdata js_1; ///< data for main joystick
128  unsigned char pad; ///< 1 byte padding
129  jsdata js_2; ///< data for partner joystick
130  unsigned char rev_lsb; ///< revision of master code lsb
131  unsigned char rev_msb; ///< revision of master code msb
132  unsigned char id; ///< message id
133  } pak; ///< access spiRxPacke as named variables
134 
135  unsigned char data[32]; ///< access spiRxPacket as an array of char
136 } spiRxPacket;
